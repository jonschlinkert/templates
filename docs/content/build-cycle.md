## Render cycle


```js
const app = new App();
// build - start

// create collections
app.task('create', cb => {
  app.create('pages');
  app.create('layouts', { type: 'layout' });
  app.create('partials', { type: 'partial' });
  cb();
});

// create collections
app.task('load', ['create'], cb => {
  // do stuff
  cb();
});

app.task('html', cb => {
  return app.toStream('pages').dest('dist')
});

// render cycle
app.task('render', ['load', 'html']);

app.build()
  .then(() => {
    console.log('done');
    // build - end
  });
```
